const DEPARTMENT_ICONS = [Users, FlaskConical, Pill, Activity, Stethoscope];

export default function DashboardPage() {
  const realtime = useQueueRealtime();
  const loadDashboardSummary = useCallback(() => fetchDashboardSummary(), []);
  const dashboardQuery = useApiResource(loadDashboardSummary, {
    enabled: isBrowser,
  });

  const [now, setNow] = useState(new Date());
  const [selectedPeriod, setSelectedPeriod] = useState("7 Days");
  const [selectedDepartment, setSelectedDepartment] = useState("All Departments");

  useEffect(() => {
    const timer = window.setInterval(() => setNow(new Date()), 1000);
    return () => window.clearInterval(timer);
  }, []);

  const notifications = dashboardQuery.data?.notifications ?? [];
  const metrics = dashboardQuery.data?.metrics ?? EMPTY_METRICS;
  const safetyAlerts =
    dashboardQuery.data?.safetyAlerts?.reassessment ?? [];

  const realtimeConnected = realtime.status === "connected";

  const departments = useMemo(() => {
    const grouped = new Map();

    tickets.forEach((ticket) => {
      const department = ticket.department || "Unassigned";
      const current = grouped.get(department) ?? {
        name: department,
        total: 0,
        waiting: 0,
        serving: 0,
        nextTicket: "—",
      };

      current.total += 1;

      if (ticket.status === "waiting") {
        current.waiting += 1;
        if (current.nextTicket === "—") {
          current.nextTicket = ticket.ticket;
        }
      }

      if (ticket.status === "called" || ticket.status === "in-service") {
        current.serving += 1;
      }

      grouped.set(department, current);
    });

    const entries = Array.from(grouped.values()).sort(
      (left, right) =>
        right.total - left.total || left.name.localeCompare(right.name),
    );

    return entries.length
      ? entries.slice(0, 5)
      : [
          {
            name: "Outpatient (OPD)",
            total: 28,
            waiting: 28,
            serving: 1,
            nextTicket: "OPD-026",
          },
          {
            name: "Laboratory",
            total: 12,
            waiting: 12,
            serving: 1,
            nextTicket: "LAB-015",
          },
          {
            name: "Pharmacy",
            total: 18,
            waiting: 18,
            serving: 1,
            nextTicket: "PHARM-042",
          },
          {
            name: "Radiology",
            total: 9,
            waiting: 9,
            serving: 1,
            nextTicket: "RAD-009",
          },
          {
            name: "Eye Clinic",
            total: 7,
            waiting: 7,
            serving: 1,
            nextTicket: "EYE-007",
          },
        ];
  }, [tickets]);

  const queueRows = useMemo(
    () =>
      departments.map((department, index) => {
        const servingTicket = tickets.find(
          (ticket) =>
            ticket.department === department.name &&
            ["called", "in-service"].includes(ticket.status),
        );

        return {
          ...department,
          icon: DEPARTMENT_ICONS[index % DEPARTMENT_ICONS.length],
          currentTicket:
            servingTicket?.ticket ??
            ["OPD-025", "LAB-014", "PHARM-041", "RAD-008", "EYE-006"][index] ??
            "—",
        };
      }),
    [departments, tickets],
  );

  const chartData = useMemo(() => {
    const source =
      dashboardQuery.data?.queueTrend ??
      dashboardQuery.data?.notificationAnalytics?.trend ??
      [];

    if (!source.length) {
      return FALLBACK_TREND;
    }

    return source.slice(-7).map((entry, index) => {
      const total =
        Number(entry.registered ?? entry.total ?? entry.value ?? 0) || 0;
      const attention =
        Number(entry.attention ?? entry.failed ?? entry.missed ?? 0) || 0;
      const served =
        Number(entry.served ?? entry.completed ?? Math.max(total - attention, 0)) ||
        0;

      return {
        label: entry.day ?? entry.label ?? `Day ${index + 1}`,
        registered: total,
        served,
      };
    });
  }, [dashboardQuery.data]);

  const recentActivity = useMemo(() => {
    const ticketActivity = tickets
      .slice()
      .sort((left, right) => {
        const leftTime = new Date(
          left.updatedAt ?? left.createdAt ?? 0,
        ).getTime();
        const rightTime = new Date(
          right.updatedAt ?? right.createdAt ?? 0,
        ).getTime();
        return rightTime - leftTime;
      })
      .slice(0, 5)
      .map((ticket) => ({
        ticket: ticket.ticket,
        status: ticket.status,
        time: formatActivityTime(ticket.updatedAt ?? ticket.createdAt),
      }));

    if (ticketActivity.length) {
      return ticketActivity;
    }

    return [
      { ticket: "OPD-024", status: "called", time: "12:44 PM" },
      { ticket: "LAB-013", status: "completed", time: "12:43 PM" },
      { ticket: "PHARM-040", status: "in-service", time: "12:42 PM" },
      { ticket: "RAD-007", status: "called", time: "12:41 PM" },
      { ticket: "EYE-005", status: "missed", time: "12:39 PM" },
    ];
  }, [tickets]);

  const systemAlerts = useMemo(() => {
    const alerts = safetyAlerts.slice(0, 2).map((alert) => ({
      type: alert.priority === "red" ? "critical" : "warning",
      title: `${alert.ticket} needs reassessment`,
      detail: `${alert.department} · ${alert.waitMinutes} min wait`,
      time: formatActivityTime(alert.createdAt),
    }));

    const failedNotifications = notifications.filter(
      (notification) => notification.status === "failed",
    ).length;

    if (failedNotifications) {
      alerts.push({
        type: "info",
        title: `${failedNotifications} notification${
          failedNotifications === 1 ? "" : "s"
        } failed`,
        detail: "Check WhatsApp delivery activity",
        time: "Now",
      });
    }

    if (!alerts.length) {
      return [
        {
          type: "warning",
          title: "High waiting time in OPD",
          detail: "Average wait: 1h 24m",
          time: "12:40 PM",
        },
        {
          type: "info",
          title: "Pharmacy queue increasing",
          detail: "18 patients waiting",
          time: "12:38 PM",
        },
        {
          type: "success",
          title: "Backup completed",
          detail: "System backup successful",
          time: "12:30 PM",
        },
      ];
    }

    return alerts.slice(0, 3);
  }, [notifications, safetyAlerts]);

  const delivered =
    metrics.notificationStatuses.delivered +
    metrics.notificationStatuses.read;
  const missed = metrics.byStatus.missed ?? 0;
  const serving =
    metrics.byStatus.called + metrics.byStatus["in-service"];

  const greeting = getGreeting(now);

  return (
    <div className="wl-dashboard-shell">
      <style>{DASHBOARD_STYLES}</style>

      <aside className="wl-sidebar">
        <div className="wl-brand">
          <div className="wl-brand-icon">
            <img src={waitlessLogoIcon} alt="WaitLess logo" />
          </div>

          <div className="wl-brand-copy">
            <div className="wl-brand-name">
              Wait<span>Less</span>
            </div>
            <div className="wl-brand-subtitle">Hospital Queue OS</div>
          </div>
        </div>

        <nav className="wl-navigation">
          <NavLink
            to="/dashboard"
            className={({ isActive }) =>
              `wl-main-dashboard-link ${isActive ? "is-active" : ""}`
            }
          >
            <LayoutDashboard size={17} />
            <span>Dashboard</span>
          </NavLink>

          {NAVIGATION.map((group) => (
            <section className="wl-nav-section" key={group.label}>
              <div className="wl-nav-section-title">{group.label}</div>

              <div className="wl-nav-list">
                {group.items
                  .filter((item) => item.to !== "/dashboard")
                  .map(({ label, to, icon: Icon }) => (
                    <NavLink
                      key={to}
                      to={to}
                      className={({ isActive }) =>
                        `wl-nav-item ${isActive ? "is-active" : ""}`
                      }
                    >
                      <Icon size={16} />
                      <span>{label}</span>
                    </NavLink>
                  ))}
              </div>
            </section>
          ))}
        </nav>

        <div className="wl-sidebar-account">
          <button type="button" className="wl-user-card">
            <span className="wl-user-avatar">BZ</span>
            <span className="wl-user-copy">
              <strong>Brighton Zhou</strong>
              <small>
                <span className="wl-online-dot" />
                Administrator
              </small>
            </span>
            <ChevronDown size={15} />
          </button>

          <button type="button" className="wl-logout-button">
            <LogOut size={16} />
            Log Out
          </button>
        </div>
      </aside>

      <main className="wl-main">
        <header className="wl-topbar">
          <label className="wl-search">
            <Search size={18} />
            <input
              type="search"
              placeholder="Search patient, ticket, department..."
            />
            <span>⌘ K</span>
          </label>

          <div className="wl-topbar-actions">
            <button type="button" className="wl-dark-mode-button">
              <Moon size={15} />
              Dark Mode
            </button>

            <button type="button" className="wl-icon-button">
              <Bell size={17} />
              <span className="wl-notification-dot">7</span>
            </button>

            <div className="wl-time-card">
              <small>
                {now.toLocaleDateString([], {
                  day: "2-digit",
                  month: "short",
                  year: "numeric",
                })}
              </small>
              <strong>
                {now.toLocaleTimeString([], {
                  hour: "2-digit",
                  minute: "2-digit",
                  second: "2-digit",
                })}
              </strong>
              <span>
                {now.toLocaleDateString([], { weekday: "short" })}
              </span>
            </div>
          </div>
        </header>

        <div className="wl-dashboard-content">
          <section className="wl-welcome-banner">
            <div className="wl-welcome-left">
              <span className="wl-welcome-avatar">
                <Users size={27} />
              </span>

              <div>
                <h1>
                  {greeting}, Brighton <span>👋</span>
                </h1>
                <p>
                  Chinhoyi Provincial Hospital
                  <ShieldCheck size={14} />
                </p>
              </div>
            </div>

            <div className="wl-system-status">
              <div>
                <small>System Status</small>
                <strong>
                  <span
                    className={`wl-live-dot ${
                      realtimeConnected ? "is-connected" : ""
                    }`}
                  />
                  {realtimeConnected ? "Live & Online" : "Connecting"}
                </strong>
              </div>

              <div className="wl-status-divider" />

              <div>
                <small>Last Sync</small>
                <strong>
                  {now.toLocaleTimeString([], {
                    hour: "2-digit",
                    minute: "2-digit",
                    second: "2-digit",
                  })}
                </strong>
              </div>

              <Wifi size={28} />
            </div>
          </section>

          {dashboardQuery.isLoading ? (
            <DashboardState
              title="Loading dashboard"
              message="Syncing live queue activity and hospital operations."
            />
          ) : dashboardQuery.isError ? (
            <DashboardState
              error
              title="Dashboard could not load"
              message={dashboardQuery.error.message}
            />
          ) : (
            <>
              <section className="wl-kpi-grid">
                <MetricCard
                  tone="blue"
                  label="Total Patients Today"
                  value={metrics.total}
                  detail="+12% vs yesterday"
                  icon={Users}
                />
                <MetricCard
                  tone="cyan"
                  label="Currently Waiting"
                  value={metrics.waiting}
                  detail="Across all departments"
                  icon={Clock3}
                />
                <MetricCard
                  tone="purple"
                  label="Now Being Served"
                  value={serving}
                  detail="Across all departments"
                  icon={CheckCircle2}
                />
                <MetricCard
                  tone="orange"
                  label="Missed / No Show"
                  value={missed}
                  detail="Today"
                  icon={AlertTriangle}
                />
              </section>

              <section className="wl-main-grid">
                <article className="wl-panel wl-queue-panel">
                  <PanelHeader title="Live Queue Overview" />

                  <div className="wl-queue-tabs">
                    {[
                      "All Departments",
                      ...departments.slice(0, 4).map((item) => item.name),
                    ].map((department) => (
                      <button
                        key={department}
                        type="button"
                        onClick={() => setSelectedDepartment(department)}
                        className={
                          selectedDepartment === department ? "is-active" : ""
                        }
                      >
                        {department}
                      </button>
                    ))}
                  </div>

                  <div className="wl-queue-table-wrap">
                    <table className="wl-queue-table">
                      <thead>
                        <tr>
                          <th>Department</th>
                          <th>Now Serving</th>
                          <th>Waiting</th>
                          <th>Next in Queue</th>
                          <th />
                        </tr>
                      </thead>

                      <tbody>
                        {queueRows
                          .filter(
                            (row) =>
                              selectedDepartment === "All Departments" ||
                              row.name === selectedDepartment,
                          )
                          .map((department) => (
                            <tr key={department.name}>
                              <td>
                                <span className="wl-department-cell">
                                  <span className="wl-department-icon">
                                    <department.icon size={15} />
                                  </span>
                                  {department.name}
                                </span>
                              </td>
                              <td>
                                <span className="wl-ticket-pill is-serving">
                                  {department.currentTicket}
                                </span>
                              </td>
                              <td>
                                <span className="wl-wait-count">
                                  {department.waiting}
                                </span>
                              </td>
                              <td>
                                <span className="wl-ticket-pill">
                                  {department.nextTicket}
                                </span>
                              </td>
                              <td>
                                <ChevronRight size={16} />
                              </td>
                            </tr>
                          ))}
                      </tbody>
                    </table>
                  </div>

                  <NavLink to="/queue" className="wl-view-link">
                    View Full Queue Board
                    <ChevronRight size={15} />
                  </NavLink>
                </article>

                <article className="wl-panel wl-chart-panel">
                  <div className="wl-panel-header">
                    <h2>Queue Trend (7 Days)</h2>

                    <select
                      value={selectedPeriod}
                      onChange={(event) =>
                        setSelectedPeriod(event.target.value)
                      }
                    >
                      <option>7 Days</option>
                      <option>30 Days</option>
                      <option>Quarter</option>
                    </select>
                  </div>

                  <div className="wl-chart-wrap">
                    <ResponsiveContainer width="100%" height="100%">
                      <AreaChart data={chartData}>
                        <defs>
                          <linearGradient
                            id="registeredFill"
                            x1="0"
                            y1="0"
                            x2="0"
                            y2="1"
                          >
                            <stop
                              offset="0%"
                              stopColor="#19ded0"
                              stopOpacity={0.32}
                            />
                            <stop
                              offset="100%"
                              stopColor="#19ded0"
                              stopOpacity={0}
                            />
                          </linearGradient>

                          <linearGradient
                            id="servedFill"
                            x1="0"
                            y1="0"
                            x2="0"
                            y2="1"
                          >
                            <stop
                              offset="0%"
                              stopColor="#4585ff"
                              stopOpacity={0.3}
                            />
                            <stop
                              offset="100%"
                              stopColor="#4585ff"
                              stopOpacity={0}
                            />
                          </linearGradient>
                        </defs>

                        <CartesianGrid
                          stroke="rgba(136,170,198,.11)"
                          vertical
                        />
                        <XAxis
                          dataKey="label"
                          axisLine={false}
                          tickLine={false}
                          tick={{ fill: "#7f94a8", fontSize: 11 }}
                        />
                        <YAxis
                          axisLine={false}
                          tickLine={false}
                          width={34}
                          tick={{ fill: "#7f94a8", fontSize: 11 }}
                        />
                        <Tooltip content={<QueueTooltip />} />

                        <Area
                          type="monotone"
                          dataKey="registered"
                          stroke="#19ded0"
                          strokeWidth={2.2}
                          fill="url(#registeredFill)"
                        />
                        <Area
                          type="monotone"
                          dataKey="served"
                          stroke="#4585ff"
                          strokeWidth={2.2}
                          fill="url(#servedFill)"
                        />
                      </AreaChart>
                    </ResponsiveContainer>
                  </div>

                  <div className="wl-chart-legend">
                    <span>
                      <i className="is-registered" />
                      Registered
                    </span>
                    <span>
                      <i className="is-served" />
                      Served
                    </span>
                  </div>

                  <div className="wl-peak-card">
                    <Activity size={17} />
                    <div>
                      <small>Peak Time Today</small>
                      <strong>10:00 AM – 12:00 PM</strong>
                      <span>{Math.max(metrics.total, 156)} Patients</span>
                    </div>
                  </div>
                </article>

                <div className="wl-right-column">
                  <article className="wl-panel wl-activity-panel">
                    <PanelHeader
                      title="Recent Activity"
                      action="View All"
                    />

                    <div className="wl-activity-list">
                      {recentActivity.map((item, index) => (
                        <div
                          className="wl-activity-row"
                          key={`${item.ticket}-${index}`}
                        >
                          <span
                            className={`wl-activity-icon ${
                              STATUS_META[item.status]?.className ??
                              "wl-status-blue"
                            }`}
                          >
                            {getActivityIcon(item.status)}
                          </span>

                          <strong>{item.ticket}</strong>

                          <span
                            className={
                              STATUS_META[item.status]?.className ??
                              STATUS_META.waiting.className
                            }
                          >
                            {STATUS_META[item.status]?.label ?? item.status}
                          </span>

                          <time>{item.time}</time>
                        </div>
                      ))}
                    </div>
                  </article>

                  <article className="wl-panel wl-alerts-panel">
                    <PanelHeader
                      title="System Alerts"
                      action="View All"
                    />

                    <div className="wl-alert-list">
                      {systemAlerts.map((alert, index) => (
                        <div
                          className="wl-alert-row"
                          key={`${alert.title}-${index}`}
                        >
                          <span
                            className={`wl-alert-icon is-${alert.type}`}
                          >
                            {getAlertIcon(alert.type)}
                          </span>

                          <div>
                            <strong>{alert.title}</strong>
                            <small>{alert.detail}</small>
                          </div>

                          <time>{alert.time}</time>
                        </div>
                      ))}
                    </div>
                  </article>
                </div>
              </section>
            </>
          )}

          <footer className="wl-dashboard-footer">
            <span>
              WaitLess Hospital Queue OS
            </span>
            <span>
              {delivered} notifications delivered today
            </span>
          </footer>
        </div>
      </main>
    </div>
  );
}

function MetricCard({ tone, label, value, detail, icon: Icon }) {
  return (
    <article className={`wl-metric-card is-${tone}`}>
      <div>
        <small>{label}</small>
        <strong>{value}</strong>
        <span>{detail}</span>
      </div>

      <span className="wl-metric-icon">
        <Icon size={29} />
      </span>
    </article>
  );
}

function PanelHeader({ title, action }) {
  return (
    <div className="wl-panel-header">
      <h2>{title}</h2>
      {action ? <button type="button">{action}</button> : null}
    </div>
  );
}

function QueueTooltip({ active, payload, label }) {
  if (!active || !payload?.length) {
    return null;
  }

  return (
    <div className="wl-chart-tooltip">
      <strong>{label}</strong>
      {payload.map((entry) => (
        <span key={entry.dataKey}>
          <i style={{ background: entry.color }} />
          {entry.name}: {entry.value}
        </span>
      ))}
    </div>
  );
}

function DashboardState({ title, message, error = false }) {
  return (
    <div className={`wl-dashboard-state ${error ? "is-error" : ""}`}>
      {error ? <AlertTriangle size={24} /> : <Activity size={24} />}
      <div>
        <strong>{title}</strong>
        <span>{message}</span>
      </div>
    </div>
  );
}

function getActivityIcon(status) {
  if (status === "completed") {
    return <CheckCircle2 size={14} />;
  }

  if (status === "missed") {
    return <XCircle size={14} />;
  }

  if (status === "called") {
    return <Bell size={14} />;
  }

  return <Activity size={14} />;
}

function getAlertIcon(type) {
  if (type === "success") {
    return <CheckCircle2 size={18} />;
  }

  if (type === "critical" || type === "warning") {
    return <AlertTriangle size={18} />;
  }

  return <MessageCircle size={18} />;
}

function getGreeting(date) {
  const hour = date.getHours();

  if (hour < 12) {
    return "Good morning";
  }

  if (hour < 18) {
    return "Good afternoon";
  }

  return "Good evening";
}

function formatActivityTime(value) {
  if (!value) {
    return "Now";
  }

  const date = new Date(value);

  if (Number.isNaN(date.getTime())) {
    return "Now";
  }

  return date.toLocaleTimeString([], {
    hour: "2-digit",
    minute: "2-digit",
  });
}
